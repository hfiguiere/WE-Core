116  template <typename T>
117  void TPTSVFilter<T>::processBlock(T* inSamples, size_t numSamples) {
118 
119  if (_mode != Parameters::FILTER_MODE.BYPASS) {
120 
121  for (size_t idx {0}; idx < numSamples; idx++) {
122  const T sample {inSamples[idx]};
123 
124  const T yH {static_cast<T>(_h * (sample - (1.0f / _Q + _g) * _s1 - _s2))};
125 
126  const T yB {static_cast<T>(_g * yH + _s1)};
127  _s1 = _g * yH + yB;
128 
129  const T yL {static_cast<T>(_g * yB + _s2)};
130  _s2 = _g * yB + yL;
131 
132  switch (_mode) {
133  case Parameters::ModeParameter::PEAK:
134  inSamples[idx] = yB * static_cast<T>(_gain);
135  break;
136 
137  case Parameters::ModeParameter::HIGHPASS:
138  inSamples[idx] = yH * static_cast<T>(_gain);
139  break;
140 
141  default:
142  inSamples[idx] = yL * static_cast<T>(_gain);
143  break;
144  }
145  }
146  }
147  }
