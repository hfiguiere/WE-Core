63  Wavetables::Wavetables() {
64 
65  // Sine wavetable
66  for (int idx = 0; idx < Wavetables::SIZE; idx++) {
67  const double radians {idx * CoreMath::DOUBLE_TAU / Wavetables::SIZE};
68 
69  // Just a conventional sine
70  _sineTable[idx] = sin(radians);
71  }
72 
73  // Square wavetable
74  for (int idx = 0; idx < Wavetables::SIZE; idx++) {
75  const double radians {(idx * CoreMath::DOUBLE_TAU / Wavetables::SIZE) + 0.32};
76 
77  // The fourier series for a square wave produces a very sharp square with some overshoot
78  // and ripple, so this actually uses slightly lower amplitudes for each harmonic than
79  // would normally be used.
80  //
81  // Because the harmonics are lower amplitude, it needs to be scaled up by 1.2 to reach
82  // a range of -1 to +1
83  _squareTable[idx] =
84  (
85  sin(radians) +
86  (0.3/1.0) * sin(3 * radians) +
87  (0.3/2.0) * sin(5 * radians) +
88  (0.3/4.0) * sin(7 * radians) +
89  (0.3/8.0) * sin(9 * radians) +
90  (0.3/16.0) * sin(11 * radians) +
91  (0.3/32.0) * sin(13 * radians)
92  ) * 1.2;
93  }
94 
95  // Saw wavetable
96  for (int idx = 0; idx < Wavetables::SIZE; idx++) {
97  const double radians {(idx * CoreMath::DOUBLE_TAU / Wavetables::SIZE) + CoreMath::DOUBLE_PI};
98 
99  // Conventional fourier series for a saw wave, scaled to fit -1 to 1
100  _sawTable[idx] =
101  (
102  sin(radians) -
103  (1.0/2.0) * sin(2 * radians) +
104  (1.0/3.0) * sin(3 * radians) -
105  (1.0/4.0) * sin(4 * radians) +
106  (1.0/6.0) * sin(5 * radians) -
107  (1.0/8.0) * sin(6 * radians) +
108  (1.0/12.0) * sin(7 * radians) -
109  (1.0/16.0) * sin(8 * radians) +
110  (1.0/24.0) * sin(9 * radians) -
111  (1.0/32.0) * sin(10 * radians) +
112  (1.0/48.0) * sin(11 * radians) -
113  (1.0/64.0) * sin(12 * radians) +
114  (1.0/96.0) * sin(13 * radians) -
115  (1.0/128.0) * sin(14 * radians)
116  ) * (2.0 / 3.0);
117  }
118 
119  // Sidechain wavetable
120  for (int idx = 0; idx < Wavetables::SIZE; idx++) {
121  const double radians {idx * CoreMath::DOUBLE_TAU / Wavetables::SIZE};
122 
123  // y = -2sin( (0.15x - 1.21)^6 ) + 1
124  _sidechainTable[idx] =
125  (
126  -2 * sin(pow(0.15 * radians - 1.21, 6)) + 1
127  );
128  }
129  }
130 }
