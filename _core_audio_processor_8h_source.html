148  void CoreAudioProcessor::getStateInformation(MemoryBlock& destData)
149  {
150  // Compile the list of parameter values
151  std::vector<float> paramValues;
152  for (const ParameterInterface& param : _paramsList) {
153  paramValues.push_back(param.getter());
154  }
155 
156  // Build the XML
157  XmlElement root("Root");
158  XmlElement *el = root.createNewChildElement("AllUserParam");
159 
160  el->addTextElement(String(_floatVectorToString(paramValues)));
161  copyXmlToBinary(root, destData);

164  void CoreAudioProcessor::setStateInformation(const void* data, int sizeInBytes)
165  {
166  std::unique_ptr<XmlElement> pRoot(getXmlFromBinary(data, sizeInBytes));
167  std::vector<float> readParamValues;
168 
169  // Parse the XML
170  if (pRoot != NULL) {
171  forEachXmlChildElement((*pRoot), pChild) {
172  if (pChild->hasTagName("AllUserParam")) {
173 
174  // Read the values into a float array
175  String sFloatCSV = pChild->getAllSubText();
176  const std::vector<float> readParamValues = _stringToFloatVector(sFloatCSV);
177 
178  // Pass each value read from XML to a setter, being careful not to go out of
179  // range on either of the vectors
180  //
181  // For this to work it relies the order not changing between plugin versions,
182  // otherwise parameter values written by an old version will be assigned to the
183  // wrong parameters in a new version (new parameters are ok though)
184  for (int idx {0}; idx < _paramsList.size(); idx++) {
185  if (idx < readParamValues.size()) {
186  _paramsList[idx].setter(readParamValues[idx]);
187  }
188  }
189  }
190  }
191  }
192  }

237  String CoreAudioProcessor::_floatVectorToString(const std::vector<float>& fData) const {
238  String result {""};

240  if (fData.size() < 1) {
241  return result;
242  }
243 
244  for (int iii {0}; iii < (fData.size() - 1); iii++) {
245  result << String(fData[iii])<<",";
246  }
247 
248  result << String(fData[fData.size() - 1]);
249 
250  return result;
