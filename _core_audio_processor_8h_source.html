243  void CoreAudioProcessor::setStateInformation(const void* data, int sizeInBytes) {
244  std::unique_ptr<juce::XmlElement> rootElement(getXmlFromBinary(data, sizeInBytes));
245 
246  // Parse the XML
247  if (rootElement != nullptr) {
248 
249  // If state was saved using an old plugin we need to migrate the XML data
250  if (rootElement->getIntAttribute("SchemaVersion", 0) < PARAMS_SCHEMA_VERSION) {
251  rootElement = _migrateParameters(std::move(rootElement));
252  }
253 
254  // Iterate through our list of parameters, restoring them from the XML attributes
255  juce::XmlElement* paramsElement = rootElement->getChildByName("Params");
256  if (paramsElement != nullptr) {
257  for (const ParameterInterface& param : _paramsList) {
258  if (paramsElement->hasAttribute(param.name)) {
259  param.setter(paramsElement->getDoubleAttribute(param.name));
260  }
261  }
262  }
263  }

371  std::unique_ptr<juce::XmlElement> CoreAudioProcessor::_migrateParameters(std::unique_ptr<juce::XmlElement> rootElement) {
372  const int schemaVersion {rootElement->getIntAttribute("SchemaVersion", 0)};
373 
374  std::unique_ptr<juce::XmlElement> retVal = std::make_unique<juce::XmlElement>("Root");
375 
376  if (schemaVersion == 0) {
377  // This is an original parameter schema - parameters are normalised values in a single string
378 
379  forEachXmlChildElement((*rootElement), childElement) {
380  if (childElement->hasTagName("AllUserParam")) {
381 
382  // Read the values into a float array
383  juce::String sFloatCSV = childElement->getAllSubText();
384  std::vector<float> readParamValues = _stringToFloatVector(sFloatCSV);
385  _migrateParamValues(readParamValues);
386 
387  std::vector<juce::String> paramNames = _provideParamNamesForMigration();
388 
389  juce::XmlElement* paramsElement = retVal->createNewChildElement("Params");
390 
391  for (int idx {0}; idx < paramNames.size(); idx++) {
392  if (idx < readParamValues.size()) {
393  paramsElement->setAttribute(paramNames[idx], readParamValues[idx]);
394  }
395  }
396  }
397  }
398  }
399 
400  return retVal;
401  }
