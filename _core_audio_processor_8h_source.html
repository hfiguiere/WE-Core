151  void CoreAudioProcessor::getStateInformation(juce::MemoryBlock& destData)
152  {
153  // Compile the list of parameter values
154  std::vector<float> paramValues;
155  for (const ParameterInterface& param : _paramsList) {
156  paramValues.push_back(param.getter());
157  }
158 
159  // Build the XML
160  juce::XmlElement root("Root");
161  juce::XmlElement *el = root.createNewChildElement("AllUserParam");
162 
163  el->addTextElement(juce::String(_floatVectorToString(paramValues)));
164  copyXmlToBinary(root, destData);

167  void CoreAudioProcessor::setStateInformation(const void* data, int sizeInBytes)
168  {
169  std::unique_ptr<juce::XmlElement> pRoot(getXmlFromBinary(data, sizeInBytes));
170 
171  // Parse the XML
172  if (pRoot != NULL) {
173  forEachXmlChildElement((*pRoot), pChild) {
174  if (pChild->hasTagName("AllUserParam")) {
175 
176  // Read the values into a float array
177  juce::String sFloatCSV = pChild->getAllSubText();
178  const std::vector<float> readParamValues = _stringToFloatVector(sFloatCSV);
179 
180  // Pass each value read from XML to a setter, being careful not to go out of
181  // range on either of the vectors
182  //
183  // For this to work it relies the order not changing between plugin versions,
184  // otherwise parameter values written by an old version will be assigned to the
185  // wrong parameters in a new version (new parameters are ok though)
186  for (int idx {0}; idx < _paramsList.size(); idx++) {
187  if (idx < readParamValues.size()) {
188  _paramsList[idx].setter(readParamValues[idx]);
189  }
190  }
191  }
192  }
193  }
194  }
