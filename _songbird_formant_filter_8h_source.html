104  template <typename T, size_t NUM_FORMANTS>
105  void SongbirdFormantFilter<T, NUM_FORMANTS>::process(T* inSamples, size_t numSamples) {
106 
107  // If the buffer we've been passed is bigger than our static internal buffer, then we need
108  // to break it into chunks
109  const size_t numBuffersRequired {static_cast<size_t>(
110  std::ceil(static_cast<double>(numSamples) / INTERNAL_BUFFER_SIZE)
111  )};
112 
113  for (size_t bufferNumber {0}; bufferNumber < numBuffersRequired; bufferNumber++) {
114 
115  // Calculate how many samples need to be processed in this chunk
116  const size_t numSamplesRemaining {numSamples - (bufferNumber * INTERNAL_BUFFER_SIZE)};
117  const size_t numSamplesToCopy {std::min(numSamplesRemaining,
118  static_cast<size_t>(INTERNAL_BUFFER_SIZE))};
119 
120  // Empty the output buffer
121  std::fill(_outputBuffer, &_outputBuffer[numSamplesToCopy], 0);
122 
123  // Get a pointer to the start of this chunk
124  T* const bufferInputStart {&inSamples[bufferNumber * INTERNAL_BUFFER_SIZE]};
125 
126  // Perform the filtering for each formant peak
127  for (size_t filterNumber {0}; filterNumber < _filters.size(); filterNumber++) {
128 
129  // Copy the input samples to the temp buffer
130  std::copy(bufferInputStart, &bufferInputStart[numSamplesToCopy], _tempInputBuffer);
131 
132  _filters[filterNumber].processBlock(_tempInputBuffer, numSamplesToCopy);
133 
134  // Add the processed samples to the output buffer
135  for (size_t iii {0}; iii < numSamplesToCopy; iii++) {
136  _outputBuffer[iii] += _tempInputBuffer[iii];
137  }
138  }
139 
140  // Write the buffer to output
141  std::copy(_outputBuffer, &_outputBuffer[numSamplesToCopy], bufferInputStart);
142  }
143  }

145  template <typename T, size_t NUM_FORMANTS>
146  bool SongbirdFormantFilter<T, NUM_FORMANTS>::setFormants(
147  const std::array<Formant, NUM_FORMANTS>& formants) {
148 
149  bool retVal {false};
150 
151  // if the correct number of formants have been supplied,
152  // apply them to each filter in turn
153  if (_filters.size() == formants.size()) {
154  retVal = true;
155 
156  for (size_t iii {0}; iii < _filters.size(); iii++) {
157  _filters[iii].setCutoff(formants[iii].frequency);
158 
159  double gainAbs = CoreMath::dBToLinear(formants[iii].gaindB);
160  _filters[iii].setGain(gainAbs);
161  }
162  }
163 
164  return retVal;
