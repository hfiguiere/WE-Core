58  void WaveViewer::paint(Graphics &g) {
59 
60  // Down sample the wave array
61  constexpr int NUM_SAMPLES {25};
62  constexpr float SCALE {0.4};
63  constexpr float MARGIN { (1 - SCALE) / 2 };
64  const float INCREMENT {static_cast<float>(Wavetables::SIZE) / NUM_SAMPLES};
65 
66  if (_waveArrayPointer != nullptr) {
67  Path p;
68 
69  for (size_t idx {0}; idx < NUM_SAMPLES; idx++) {
70  // Calculate the index of the sample accounting for downsampling and phase shift
71  const int sampleIdx {(
72  (static_cast<int>(idx * INCREMENT + _phaseShift) % Wavetables::SIZE)
73  )};
74 
75  // Get the sample for this value
76  const double sample {_waveArrayPointer[sampleIdx] * _depth * (_isInverted ? -1 : 1)};
77 
78  // Invert the wave and scale to the height of this component
79  const double sampleX {(static_cast<double>(idx) / NUM_SAMPLES) * getWidth()};
80  const double sampleY = (0.5 - sample) * getHeight() * SCALE + getHeight() * MARGIN;
81 
82  // Add it to the path
83  if (idx == 0) {
84  p.startNewSubPath(0, sampleY);
85  } else {
86  p.lineTo(sampleX, sampleY);
87  }
88  }
89 
90  g.setColour(findColour(highlightColourId));
91  g.strokePath(p, PathStrokeType(3.0f));
92  }
93  }
