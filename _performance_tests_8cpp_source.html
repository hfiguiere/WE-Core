66  inline Stats calcAverageAndDeviation(const std::vector<double>& executionTimes) {
67  Stats retVal;
68 
69  // calculate the average first
70  retVal.average = 0;
71  for (double time : executionTimes) {
72  retVal.average += time;
73  }
74  retVal.average = retVal.average / static_cast<double>(executionTimes.size());
75 
76  // now calculate deviation
77  retVal.deviation = 0;
78  for (const double& time : executionTimes) {
79  retVal.deviation += std::pow((time - retVal.average), 2);
80  }
81  retVal.deviation = retVal.deviation / (static_cast<double>(executionTimes.size()) - 1);
82  retVal.deviation = std::sqrt(retVal.deviation);
83 
84  return retVal;
85  }

108 SCENARIO("Performance: CarveDSPUnit, 100 buffers of 1024 samples each") {
109  GIVEN("A CarveDSPUnit and a buffer of samples") {
110 
111  const int NUM_BUFFERS {100};
112  std::vector<double> buffer(1024);
113  WECore::Carve::CarveDSPUnit<double> mCarve;
114 
115  // set the performance limits
116  Limits mLimits{0.11, 0.07, 0.01};
117 
118  // store the execution time for each buffer
119  std::vector<double> executionTimes;
120 
121  WHEN("The samples are processed") {
122  // turn the unit on
123  mCarve.setMode(2);
124 
125  for (int nbuf {0}; nbuf < NUM_BUFFERS; nbuf++) {
126 
127  // fill the buffer with a sine wave
128  int iii {0};
129  std::generate(buffer.begin(), buffer.end(), [&iii]{ return std::sin(iii++); });
130 
131 
132  // do processing
133  const clock_t startTime {clock()};
134  for (size_t jjj {0}; jjj < buffer.size(); jjj++) {
135  buffer[jjj] = mCarve.process(buffer[jjj]);
136  }
137  const clock_t endTime {clock()};
138 
139  // calculate the execution time
140  const double CLOCKS_PER_MICROSEC {static_cast<double>(CLOCKS_PER_SEC) / 1000};
141  const double executionTime {static_cast<double>(endTime - startTime) / CLOCKS_PER_MICROSEC};
142  executionTimes.push_back(executionTime);
143  CHECK(executionTime < mLimits.INDIVIDUAL);
144  }
145 
146  THEN("The average and variance are within limits") {
147  Stats mStats = calcAverageAndDeviation(executionTimes);
148  CHECK(mStats.average < mLimits.AVERAGE);
149  CHECK(mStats.deviation < mLimits.DEVIATION);
150 
151  appendToResultsFile(mStats, Catch::getResultCapture().getCurrentTestName());
152  }
153  }
154  }
