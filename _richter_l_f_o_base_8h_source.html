255  void RichterLFOBase::_calcPhaseOffset(double timeInSeconds) {
256  if (_phaseSyncSwitch && _needsPhaseCalc) {
257  static double waveLength {1 / _freq};
258  static double waveTimePosition {0};
259 
260  if (waveLength < timeInSeconds) {
261  waveTimePosition = fmod(timeInSeconds, waveLength);
262  } else {
263  waveTimePosition = timeInSeconds;
264  }
265  _indexOffset = static_cast<int>(waveTimePosition / waveLength) * Wavetables::SIZE + _manualPhase;
266  }
267 
268  if (!_phaseSyncSwitch && _needsPhaseCalc) {
269  _indexOffset = _manualPhase;
270  }
271  _needsPhaseCalc = false;
272 
273  }

276  void RichterLFOBase::_calcFreq(double bpm) {
277  // calculate the frequency based on whether tempo sync is active
278 
279  _tempoFreq = (bpm / 60) * (_tempoDenom / _tempoNumer);
280 
281  if (_tempoSyncSwitch) { _freq = _tempoFreq; }
282 
283  _freq = Parameters::FREQ.BoundsCheck(_freq);
284 
285  }

287  void RichterLFOBase::calcIndexAndScaleInLoop() {
288  // calculate the current index within the wave table
289  _index = static_cast<int>(static_cast<long>(_samplesProcessed * static_cast<long double>(_currentScale)) % Wavetables::SIZE);
290 
291  if ((!CoreMath::compareFloatsEqual(_nextScale, _currentScale)) && (_index == 0)) {
292  _currentScale = _nextScale;
293  _samplesProcessed = 0;
294  }
295 
296 
297  // Must provide two possibilities for each index lookup in order to protect the array from
298  // being overflowed by the indexOffset, the first if statement uses the standard index
299  // lookup while second if statement deals with the overflow possibility
300 
301  if ((_index + _indexOffset) < Wavetables::SIZE) {
302  _gain = _waveArrayPointer[_index + _indexOffset];
303  } else if ((_index + _indexOffset) >= Wavetables::SIZE) {
304  _gain = _waveArrayPointer[(_index + _indexOffset) % Wavetables::SIZE];
305  }
306 
307  _samplesProcessed++;
308  }
