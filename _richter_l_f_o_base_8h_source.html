249  void RichterLFOBase::_calcPhaseOffset(double timeInSeconds) {
250  if (_phaseSyncSwitch && _needsPhaseCalc) {
251  static double waveLength {1 / _freq};
252  static double waveTimePosition {0};
253 
254  if (waveLength < timeInSeconds) {
255  waveTimePosition = fmod(timeInSeconds, waveLength);
256  } else {
257  waveTimePosition = timeInSeconds;
258  }
259  _indexOffset = static_cast<int>(waveTimePosition / waveLength) * Parameters::kWaveArraySize + _manualPhase;
260  }
261 
262  if (!_phaseSyncSwitch && _needsPhaseCalc) {
263  _indexOffset = _manualPhase;
264  }
265  _needsPhaseCalc = false;
266 
267  }

270  void RichterLFOBase::_calcFreq(double bpm) {
271  // calculate the frequency based on whether tempo sync is active
272 
273  _tempoFreq = (bpm / 60) * (_tempoDenom / _tempoNumer);

275  if (_tempoSyncSwitch) { _freq = _tempoFreq; }

281  void RichterLFOBase::calcIndexAndScaleInLoop() {
282  // calculate the current index within the wave table
283  _index = static_cast<int>(static_cast<long>(_samplesProcessed * static_cast<long double>(_currentScale)) % Parameters::kWaveArraySize);
284 
285  if ((!CoreMath::compareFloatsEqual(_nextScale, _currentScale)) && (_index == 0)) {
286  _currentScale = _nextScale;
287  _samplesProcessed = 0;
288  }
289 
290 
291  // Must provide two possibilities for each index lookup in order to protect the array from being overflowed by the indexOffset, the first if statement uses the standard index lookup while second if statement deals with the overflow possibility
292 
293  if ((_index + _indexOffset) < Parameters::kWaveArraySize) {
294  _gain = _waveArrayPointer[_index + _indexOffset];
295  } else if ((_index + _indexOffset) >= Parameters::kWaveArraySize) {
296  _gain = _waveArrayPointer[(_index + _indexOffset) % Parameters::kWaveArraySize];
297  }
298 
299  _samplesProcessed++;
300  }
