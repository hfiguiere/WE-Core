272  template <typename T>
273  void MONSTRBand<T>::process2in2out(T* leftSample, T* rightSample, size_t numSamples) {
274 
275  // Apply the filtering before processing
276  _filterSamples(leftSample, rightSample, static_cast<int>(numSamples));
277 
278  if (_isActive) {
279  // Do the actual stereo widening or narrowing
280  // Based on: http://musicdsp.org/showArchiveComment.php?ArchiveID=256
281  double coef_S {_width * 0.5};
282 
283  for (size_t iii {0}; iii < numSamples; iii++) {
284 
285  double mid {(leftSample[iii] + rightSample[iii]) * 0.5};
286  double side {(rightSample[iii] - leftSample[iii]) * coef_S};
287 
288  leftSample[iii] = mid - side;
289  rightSample[iii] = mid + side;
290  }
291  }
292  }

294  template <typename T>
295  void MONSTRBand<T>::_filterSamples(T* inLeftSamples, T* inRightSamples, int numSamples) {
296  T* channelsArray[2];
297 
298  channelsArray[0] = inLeftSamples;
299  channelsArray[1] = inRightSamples;
300 
301  if (_isLower) {
302  _highCut1.process(numSamples, channelsArray);
303  _highCut2.process(numSamples, channelsArray);
304  } else if (_isUpper) {
305  _lowCut1.process(numSamples, channelsArray);
306  _lowCut2.process(numSamples, channelsArray);
307  } else {
308  _lowCut1.process(numSamples, channelsArray);
309  _lowCut2.process(numSamples, channelsArray);
310  _highCut1.process(numSamples, channelsArray);
311  _highCut2.process(numSamples, channelsArray);
312  }
313  }
