200  template <typename SampleType>
201  void MONSTRBand<SampleType>::setLowCutoff(double val) {
202  // if this is the lowest band, then do not cut the low frequencies
203  if (_bandType == BandType::MIDDLE) {
204  _lowCutoffHz = Parameters::CROSSOVERLOWER.BoundsCheck(val);
205  _lowCut1.setup(FILTER_ORDER, _sampleRate, _lowCutoffHz);
206  _lowCut2.setup(FILTER_ORDER, _sampleRate, _lowCutoffHz);
207  } else if (_bandType == BandType::UPPER) {
208  _lowCutoffHz = Parameters::CROSSOVERUPPER.BoundsCheck(val);
209  _lowCut1.setup(FILTER_ORDER, _sampleRate, _lowCutoffHz);
210  _lowCut2.setup(FILTER_ORDER, _sampleRate, _lowCutoffHz);
211  }

241  template <typename SampleType>
242  void MONSTRBand<SampleType>::_filterSamples(SampleType* inLeftSamples, SampleType* inRightSamples, int numSamples) {
243  SampleType* channelsArray[2];
244 
245  channelsArray[0] = inLeftSamples;
246  channelsArray[1] = inRightSamples;
247 
248  if (_bandType == BandType::LOWER) {
249  _highCut1.process(numSamples, channelsArray);
250  _highCut2.process(numSamples, channelsArray);
251  } else if (_bandType == BandType::UPPER) {
252  _lowCut1.process(numSamples, channelsArray);
253  _lowCut2.process(numSamples, channelsArray);
254  } else {
255  _lowCut1.process(numSamples, channelsArray);
256  _lowCut2.process(numSamples, channelsArray);
257  _highCut1.process(numSamples, channelsArray);
258  _highCut2.process(numSamples, channelsArray);
259  }
260  }
