232  template <typename SampleType>
233  void MONSTRBand<SampleType>::setLowCutoff(double val) {
234  // if this is the lowest band, then do not cut the low frequencies
235  if (!_isLower && !_isUpper) {
236  _lowCutoffHz = Parameters::CROSSOVERLOWER.BoundsCheck(val);
237  _lowCut1.setup(FILTER_ORDER, _sampleRate, _lowCutoffHz);
238  _lowCut2.setup(FILTER_ORDER, _sampleRate, _lowCutoffHz);
239  } else if (_isUpper) {

241  _lowCut1.setup(FILTER_ORDER, _sampleRate, _lowCutoffHz);

273  template <typename SampleType>
274  void MONSTRBand<SampleType>::_filterSamples(SampleType* inLeftSamples, SampleType* inRightSamples, int numSamples) {
275  SampleType* channelsArray[2];
276 
277  channelsArray[0] = inLeftSamples;
278  channelsArray[1] = inRightSamples;
279 
280  if (_isLower) {
281  _highCut1.process(numSamples, channelsArray);
282  _highCut2.process(numSamples, channelsArray);
283  } else if (_isUpper) {
284  _lowCut1.process(numSamples, channelsArray);
285  _lowCut2.process(numSamples, channelsArray);
286  } else {
287  _lowCut1.process(numSamples, channelsArray);
288  _lowCut2.process(numSamples, channelsArray);
289  _highCut1.process(numSamples, channelsArray);
290  _highCut2.process(numSamples, channelsArray);
291  }
