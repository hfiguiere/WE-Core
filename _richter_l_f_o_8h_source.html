280  void RichterLFO::_calcPhaseOffset(double timeInSeconds) {
281  if (_phaseSyncSwitch && _needsPhaseCalc) {
282  static double waveLength {1 / _freq};
283  static double waveTimePosition {0};
284 
285  if (waveLength < timeInSeconds) {
286  waveTimePosition = fmod(timeInSeconds, waveLength);
287  } else {
288  waveTimePosition = timeInSeconds;
289  }
290  _indexOffset = static_cast<int>(waveTimePosition / waveLength) * Wavetables::SIZE + _manualPhase;
291  }
292 
293  if (!_phaseSyncSwitch && _needsPhaseCalc) {
294  _indexOffset = _manualPhase;
295  }
296  _needsPhaseCalc = false;
297 
298  }

300  void RichterLFO::_calcFreq(double bpm) {
301  // calculate the frequency based on whether tempo sync is active
302 
303  _tempoFreq = (bpm / 60) * (_tempoDenom / _tempoNumer);
304 
305  if (_tempoSyncSwitch) { _freq = _tempoFreq; }
306 
307  _freq = Parameters::FREQ.BoundsCheck(_freq);
308 
309  }

311  void RichterLFO::calcIndexAndScaleInLoop() {
312  // calculate the current index within the wave table
313  _index = static_cast<int>(static_cast<long>(_samplesProcessed * static_cast<long double>(_currentScale)) % Wavetables::SIZE);
314 
315  if ((!CoreMath::compareFloatsEqual(_nextScale, _currentScale)) && (_index == 0)) {
316  _currentScale = _nextScale;
317  _samplesProcessed = 0;
318  }
319 
320 
321  // Must provide two possibilities for each index lookup in order to protect the array from
322  // being overflowed by the indexOffset, the first if statement uses the standard index
323  // lookup while second if statement deals with the overflow possibility
324 
325  if ((_index + _indexOffset) < Wavetables::SIZE) {
326  _gain = _waveArrayPointer[_index + _indexOffset];
327  } else if ((_index + _indexOffset) >= Wavetables::SIZE) {
328  _gain = _waveArrayPointer[(_index + _indexOffset) % Wavetables::SIZE];
329  }
330 
331  _samplesProcessed++;
332  }

334  double RichterLFO::_getNextOutputImpl(double /*inSample*/) {
335  calcIndexAndScaleInLoop();
336 
337  // Get the mod amount to use, divide by 2 to reduce range to -0.5:0.5
338  const double modAmount {_modulationSource != nullptr ? _modulationSource->getNextOutput(0) / 2 : 0};
339 
340  _calcFreqInLoop(modAmount);
341  _calcDepthInLoop(modAmount);
342 
343  if (_bypassSwitch) {
344  // Produce a value in the range -1:1, invert if needed
345  return (_gain * _depth) * (_invertSwitch ? -1 : 1);
346  } else {
347  return 0;
348  }
349  }

351  void RichterLFO::_calcFreqInLoop(double modAmount) {
352  // calculate the frequency based on whether tempo sync or frequency modulation is active
353 
354  if (!_tempoSyncSwitch) {
355  _freq = _rawFreq + (_freqMod * (Parameters::FREQ.maxValue / 2) * modAmount);
356  }
357 
358  // Bounds check frequency after the modulation is applied to it
359  _freq = Parameters::FREQ.BoundsCheck(_freq);
360 
361  }
362 
363  void RichterLFO::_calcDepthInLoop(double modAmount) {
364  // Check whether MOD oscs are activated and apply depth parameter modulation accordingly
365  _depth = _rawDepth + (_depthMod * Parameters::DEPTH.maxValue * modAmount);
366  _depth = Parameters::DEPTH.BoundsCheck(_depth);
367  }
