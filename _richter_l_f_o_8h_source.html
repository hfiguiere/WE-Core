266  void RichterLFO::_calcPhaseOffset(double timeInSeconds) {
267  if (_phaseSyncSwitch && _needsPhaseCalc) {
268  double waveLength {1 / _freq};
269  double waveTimePosition {0};
270 
271  if (waveLength < timeInSeconds) {
272  waveTimePosition = fmod(timeInSeconds, waveLength);
273  } else {
274  waveTimePosition = timeInSeconds;
275  }
276  _indexOffset = static_cast<int>(waveTimePosition / waveLength) * Wavetables::SIZE + _manualPhase;
277  }
278 
279  if (!_phaseSyncSwitch && _needsPhaseCalc) {
280  _indexOffset = _manualPhase;
281  }
282  _needsPhaseCalc = false;
283  }

285  void RichterLFO::_calcFreq(double bpm) {
286  // calculate the frequency based on whether tempo sync is active
287 
288  const double tempoFreq {(bpm / 60) * (_tempoDenom / _tempoNumer)};
289 
290  if (_tempoSyncSwitch) {
291  _freq = tempoFreq;
292  }
293 
294  _freq = Parameters::FREQ.BoundsCheck(_freq);
295  }

297  void RichterLFO::calcIndexAndScaleInLoop() {
298  // calculate the current index within the wave table
299  _index = static_cast<int>(static_cast<long>(_samplesProcessed * static_cast<long double>(_currentScale)) % Wavetables::SIZE);
300 
301  if ((!CoreMath::compareFloatsEqual(_nextScale, _currentScale)) && (_index == 0)) {
302  _currentScale = _nextScale;
303  _samplesProcessed = 0;
304  }
305 
306 
307  // Must provide two possibilities for each index lookup in order to protect the array from
308  // being overflowed by the indexOffset, the first if statement uses the standard index
309  // lookup while second if statement deals with the overflow possibility
310 
311  if ((_index + _indexOffset) < Wavetables::SIZE) {
312  _gain = _waveArrayPointer[_index + _indexOffset];
313  } else if ((_index + _indexOffset) >= Wavetables::SIZE) {
314  _gain = _waveArrayPointer[(_index + _indexOffset) % Wavetables::SIZE];
315  }
316 
317  _samplesProcessed++;
318  }

320  double RichterLFO::_getNextOutputImpl(double /*inSample*/) {
321  calcIndexAndScaleInLoop();
322 
323  // Get the mod amount to use, divide by 2 to reduce range to -0.5:0.5
324  const double modAmount {_modulationSource != nullptr ? _modulationSource->getNextOutput(0) / 2 : 0};
325 
326  _calcFreqInLoop(modAmount);
327 
328  // Calculate the depth value after modulation
329  const double depth {Parameters::DEPTH.BoundsCheck(
330  _rawDepth + (_depthMod * Parameters::DEPTH.maxValue * modAmount)
331  )};
332 
333  if (_bypassSwitch) {
334  // Produce a value in the range -1:1, invert if needed
335  return (_gain * depth) * (_invertSwitch ? -1 : 1);
336  } else {
337  return 0;
338  }
339  }
340 
341  void RichterLFO::_calcFreqInLoop(double modAmount) {
342  // calculate the frequency based on whether tempo sync or frequency modulation is active
343 
344  if (!_tempoSyncSwitch) {
345  _freq = _rawFreq + (_freqMod * (Parameters::FREQ.maxValue / 2) * modAmount);
346  }
347 
348  // Bounds check frequency after the modulation is applied to it
349  _freq = Parameters::FREQ.BoundsCheck(_freq);
350  }
