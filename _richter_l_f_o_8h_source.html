157  void RichterLFO::_calcFreqInLoop(int modBypassSwitch, double modGain) {
158  // calculate the frequency based on whether tempo sync or frequency modulation is active
159 
160  if (!_tempoSyncSwitch) {
161  if (modBypassSwitch) {
162  _freq = _rawFreq + (_freqMod * (Parameters::FREQ.maxValue / 2) * modGain);
163  } else {
164  _freq = _rawFreq;
165  }
166  }
167 
168  // Bounds check frequency after the modulation is applied to it
169  _freq = Parameters::FREQ.BoundsCheck(_freq);
170 
171  }

173  void RichterLFO::_calcDepthInLoop(int modBypassSwitch, double modGain) {
174  // Check whether MOD oscs are activated and apply depth parameter modulation accordingly
175 
176  if (modBypassSwitch) {
177  _depth = _rawDepth + (_depthMod * Parameters::DEPTH.maxValue * modGain);
178  } else {
179  _depth = _rawDepth;
180  }
181 
182  _depth = Parameters::DEPTH.BoundsCheck(_depth);
183 
184  }

186  double RichterLFO::_calcGain(int modBypassSwitch, double modGain) {
187  _calcFreqInLoop(modBypassSwitch, modGain);
188  _calcDepthInLoop(modBypassSwitch, modGain);
189 
190  if (_bypassSwitch) {
191  // Convert range from -1:1 to 0:1
192  return 1 + (_gain * _depth / 2 - 0.5) + (1 - _depth) / 2;
193  } else {
194  return 1;
195  }
