148  void CoreLookAndFeel::drawRotarySlider(Graphics& g,
149  int /*x*/,
150  int /*y*/,
151  int width,
152  int height,
153  float /*sliderPosProportional*/,
154  float /*rotaryStartAngle*/,
155  float /*rotaryEndAngle*/,
156  Slider &slider) {
157 
158  // calculate useful constants
159  const double rangeOfMotion {260 * (CoreMath::DOUBLE_PI / 180)};
160  const double rotation {((slider.getValue() - slider.getMinimum()) / (slider.getMaximum() - slider.getMinimum())) * rangeOfMotion - rangeOfMotion / 2};
161 
162  const int margin {1};
163  const float diameter {static_cast<float>(height - margin * 2)};
164 
165  // draw centre circle
166  Path p;
167  g.setColour(darkColour);
168  p.addEllipse(width / 2 - diameter / 2, height / 2 - diameter / 2, diameter, diameter);
169  g.fillPath(p);
170 
171  // draw outer ring
172  if (slider.isEnabled()) {
173  g.setColour(highlightColour);
174  } else {
175  g.setColour(lightColour);
176  }
177 
178  p.clear();
179 
180  const double gap {0.4};
181  p.addCentredArc(width / 2, height / 2, diameter / 2, diameter / 2, rotation, gap, 2 * CoreMath::DOUBLE_PI - gap, true);
182 
183  g.strokePath(p, PathStrokeType(2.0f));
184  }

186  void CoreLookAndFeel::drawLinearSliderThumb(Graphics& g,
187  int x,
188  int y,
189  int width,
190  int height,
191  float sliderPos,
192  float /*minSliderPos*/,
193  float /*maxSliderPos*/,
194  const Slider::SliderStyle style,
195  Slider& slider) {
196 
197  const float sliderRadius = static_cast<float>(getSliderThumbRadius(slider) - 2);
198 
199  Colour* ring;
200 
201  if (slider.isEnabled()) {
202  ring = &highlightColour;
203  } else {
204  ring = &lightColour;
205  }
206 
207  if (style == Slider::LinearHorizontal || style == Slider::LinearVertical)
208  {
209  float kx, ky;
210 
211  if (style == Slider::LinearVertical)
212  {
213  kx = x + width * 0.5f;
214  ky = sliderPos;
215  }
216  else
217  {
218  kx = sliderPos;
219  ky = y + height * 0.5f;
220  }
221 
222  Path p;
223  p.addEllipse(kx - sliderRadius, ky - sliderRadius, sliderRadius * 2, sliderRadius * 2);
224 
225  g.setColour(darkColour);
226  g.fillPath(p);
227 
228  g.setColour(*ring);
229  g.strokePath(p, PathStrokeType(2.0f));
230  }

319  buttonX + buttonW * arrowX, buttonY + buttonH * 0.45f);
320 
321  p.addTriangle(buttonX + buttonW * 0.5f, buttonY + buttonH * (0.55f + arrowH),
322  buttonX + buttonW * (1.0f - arrowX), buttonY + buttonH * 0.55f,
323  buttonX + buttonW * arrowX, buttonY + buttonH * 0.55f);
324 
325  g.setColour(box.isPopupActive() ? highlightColour : lightColour);
