227  template <typename SampleType>
228  void SimpleCompressor<SampleType>::process1in1out(SampleType* inSamples,
229  size_t numSamples) {
230 
231  for (size_t index {0}; index < numSamples; index++) {
232 
233  // Rectify the input and convert to dB
234  const SampleType absdB {static_cast<SampleType>(
235  CoreMath::linearTodB(std::abs(inSamples[index]))
236  )};
237 
238  // Compute gain
239  const SampleType gainComp {_computeGain(absdB)};
240 
241  // Take the difference and pass to the level detection
242  const SampleType level {_computeLevel(absdB - gainComp)};
243 
244  // Convert to linear and apply the gain
245  _gainApplied = static_cast<SampleType>(CoreMath::dBToLinear(-level));
246 
247  inSamples[index] = inSamples[index] * _gainApplied;
248  }
249  }

264  template <typename SampleType>
265  SampleType SimpleCompressor<SampleType>::_computeLevel(SampleType inSample) {
266 
267  if (inSample > _levelDetectorState) {
268  _levelDetectorState = _attackCoef * _levelDetectorState + (1 - _attackCoef) * inSample;
269  } else {
270  _levelDetectorState = _releaseCoef * _levelDetectorState + (1 - _releaseCoef) * inSample;
271  }
272 
273  return _levelDetectorState;
274  }

276  template <typename SampleType>
277  SampleType SimpleCompressor<SampleType>::_computeGain(SampleType inSample) {
278 
279  SampleType retVal {inSample};
280 
281  if (_direction == Direction::DOWNWARD) {
282 
283  // Downward compression
284  if (inSample - _threshold < -_kneeWidth / 2) {
285  // Level is below threshold and knee - do nothing
286 
287  } else if (std::abs(inSample - _threshold) <= (_kneeWidth / 2)) {
288  // Level is within the range of the knee
289  retVal = inSample + (1 / _ratio - 1)
290  * std::pow(inSample - _threshold + (_kneeWidth / 2), 2);
291 
292  } else {
293  // Level is above threshold and knee
294  retVal = _threshold + (inSample - _threshold) / _ratio;
295  }
296 
297  } else {
298 
299  // Upward compression
300  if (inSample - _threshold < -_kneeWidth / 2) {
301  // Level is below theshold and knee
302  retVal = _threshold - (_threshold - inSample) / _ratio;
303 
304  } else if (std::abs(inSample - _threshold) <= (_kneeWidth / 2)) {
305  // Level is within the range of the knee
306  retVal = inSample + (1 - 1 / _ratio) \
307  * std::pow(inSample - _threshold - _kneeWidth / 2, 2) / (2 * _kneeWidth);
308 
309  } else {
310  // Level is above threshold and knee - do nothing
311  }
312 
313  }
314 
315  return retVal;
316  }
