122  GIVEN("A SongbirdFilterModule and a buffer of sine samples") {
123  std::vector<double> leftBuffer(1024);
124  std::vector<double> rightBuffer(1024);
125  const std::vector<double>& expectedOutputLeft =
126  TestData::Songbird::Data.at(Catch::getResultCapture().getCurrentTestName() + "-left");
127  const std::vector<double>& expectedOutputRight =
128  TestData::Songbird::Data.at(Catch::getResultCapture().getCurrentTestName() + "-right");
129 
130  WECore::Songbird::SongbirdFilterModule<double> mSongbird;
131 
132  // Set some parameters for the input signal
133  constexpr size_t SAMPLE_RATE {44100};
134  constexpr size_t SINE_FREQ {1000};
135  constexpr double SAMPLES_PER_CYCLE {SAMPLE_RATE / SINE_FREQ};
136 
137  // fill the buffers, phase shift the right one so that they're not identical
138  std::generate(leftBuffer.begin(),
139  leftBuffer.end(),
140  [iii = 0]() mutable {return std::sin(CoreMath::LONG_TAU * (iii++ / SAMPLES_PER_CYCLE));} );
141  std::generate(rightBuffer.begin(),
142  rightBuffer.end(),
143  [iii = 0]() mutable {return std::sin(CoreMath::LONG_TAU * (iii++ / SAMPLES_PER_CYCLE) + CoreMath::LONG_PI);} );
144  WHEN("The parameters are set and samples are processed") {
145  // Set freq mode
146  mSongbird.setVowel1(1);
147  mSongbird.setVowel2(2);
148  mSongbird.setModMode(true);
149 
150  mSongbird.Process2in2out(&leftBuffer[0], &rightBuffer[0], leftBuffer.size());
151 
152  THEN("The output is as expected") {
153  for (size_t iii {0}; iii < leftBuffer.size(); iii++) {
154  CHECK(leftBuffer[iii] == Approx(expectedOutputLeft[iii]).margin(0.00001));
