150  template <typename T>
151  void MONSTRCrossover<T>::Process2in2out(T* leftSample, T* rightSample, size_t numSamples) {
152 
153  // If the buffer we've been passed is bigger than our static internal buffer, then we need
154  // to break it into chunks
155  const size_t numBuffersRequired {static_cast<size_t>(
156  std::ceil(static_cast<double>(numSamples) / INTERNAL_BUFFER_SIZE)
157  )};
158 
159  for (size_t bufferNumber {0}; bufferNumber < numBuffersRequired; bufferNumber++) {
160 
161  // Calculate how many samples need to be processed in this chunk
162  const size_t numSamplesRemaining {numSamples - (bufferNumber * INTERNAL_BUFFER_SIZE)};
163  const size_t numSamplesToCopy {std::min(numSamplesRemaining,
164  static_cast<size_t>(INTERNAL_BUFFER_SIZE))};
165 
166  T* const leftBufferInputStart {&leftSample[bufferNumber * INTERNAL_BUFFER_SIZE]};
167  T* const rightBufferInputStart {&rightSample[bufferNumber * INTERNAL_BUFFER_SIZE]};
168 
169  std::copy(leftBufferInputStart, &leftBufferInputStart[numSamplesToCopy], _band1LeftBuffer);
170  std::copy(rightBufferInputStart, &rightBufferInputStart[numSamplesToCopy], _band1RightBuffer);
171  std::copy(leftBufferInputStart, &leftBufferInputStart[numSamplesToCopy], _band2LeftBuffer);
172  std::copy(rightBufferInputStart, &rightBufferInputStart[numSamplesToCopy], _band2RightBuffer);
173  std::copy(leftBufferInputStart, &leftBufferInputStart[numSamplesToCopy], _band3LeftBuffer);
174  std::copy(rightBufferInputStart, &rightBufferInputStart[numSamplesToCopy], _band3RightBuffer);
175 
176  // let each band do its processing
177  band1.process2in2out(_band1LeftBuffer, _band1RightBuffer, numSamplesToCopy);
178  band2.process2in2out(_band2LeftBuffer, _band2RightBuffer, numSamplesToCopy);
179  band3.process2in2out(_band3LeftBuffer, _band3RightBuffer, numSamplesToCopy);
180 
181  // combine the output from each band, and write to output
182  for (size_t iii {0}; iii < numSamplesToCopy; iii++) {
183  leftBufferInputStart[iii] = _band1LeftBuffer[iii]
184  + _band2LeftBuffer[iii]
185  + _band3LeftBuffer[iii];
186 
187  rightBufferInputStart[iii] = _band1RightBuffer[iii]
188  + _band2RightBuffer[iii]
189  + _band3RightBuffer[iii];
190  }
191  }
192  }
