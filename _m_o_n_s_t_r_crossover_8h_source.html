276  template <typename SampleType>
277  void MONSTRCrossover<SampleType>::setCrossoverFrequency(size_t index, double val) {
278 
279  if (index < _bands.size() - 1) {
280 
281  // Set the crossover frequency
282  val = Parameters::CROSSOVER_FREQUENCY.BoundsCheck(val);
283  _bands[index].band.setHighCutoff(val);
284  _bands[index + 1].band.setLowCutoff(val);
285 
286  // Make sure the crossover frequencies are still in the correct order
287  for (size_t otherCrossoverIndex {0}; otherCrossoverIndex < _bands.size() - 1; otherCrossoverIndex++) {
288 
289  const double otherCrossoverFrequency {getCrossoverFrequency(otherCrossoverIndex)};
290 
291  const bool needsCrossoverUpdate {
292  // We've moved the crossover frequency of index below another one that should be
293  // below it - move the other one to the new value
294  (val < otherCrossoverFrequency && otherCrossoverIndex < index) ||
295 
296  // We've moved the crossover frequency of index above another one that should be
297  // above it - move the other one to the new value
298  (otherCrossoverFrequency > val && index > otherCrossoverIndex)
299  };
300 
301  if (needsCrossoverUpdate) {
302  // We've moved the crossover frequency of index below another one that should be
303  // below it - move the other one to the new value
304  setCrossoverFrequency(otherCrossoverIndex, val);
305 
306  }
307  }
308  }

370  template <typename SampleType>
371  void MONSTRCrossover<SampleType>::addBand() {
372 
373  if (static_cast<int>(_numBands) < Parameters::NUM_BANDS.maxValue) {
374 
375  // Convert the current highest band to a middle band
376  _bands[_numBands - 1].band.setBandType(BandType::MIDDLE);
377 
378  // Set the next band up to be an upper band
379  _bands[_numBands].band.setBandType(BandType::UPPER);
380 
381  const double oldHighestCrossover {getCrossoverFrequency(_numBands - 2)};
382 
383  if (oldHighestCrossover < Parameters::CROSSOVER_FREQUENCY.maxValue) {
384  // The old highest crossover frequency is below the maximum, insert the new one halfway
385  // between it and the maximum
386  const double topBandWidth {Parameters::CROSSOVER_FREQUENCY.maxValue - oldHighestCrossover};
387  setCrossoverFrequency(_numBands - 1, oldHighestCrossover + (topBandWidth / 2));
388  } else {
389  // The old highest crossover is at the maximum, move it halfway down to the one below
390  // and place the new one at the maximum
391  const double secondHighestCrossover {getCrossoverFrequency(_numBands - 3)};
392  const double bandWidth {Parameters::CROSSOVER_FREQUENCY.maxValue - secondHighestCrossover};
393 
394  setCrossoverFrequency(_numBands - 2, secondHighestCrossover + (bandWidth / 2));
395  setCrossoverFrequency(_numBands - 1, Parameters::CROSSOVER_FREQUENCY.maxValue);
396  }
397 
398  _numBands++;
399  }
400 
401  reset();
402  }

405  void MONSTRCrossover<SampleType>::removeBand() {
406 
407  if (static_cast<int>(_numBands) > Parameters::NUM_BANDS.minValue) {
408  // Make sure the band's parameters are reset to their default values
409  setIsActive(_numBands - 1, true);
410  setIsMuted(_numBands - 1, false);
411  setEffectsProcessor(_numBands - 1, nullptr);
412 
413  // Decrement the counter and make the new highest band the upper band
414  _numBands--;
415  _bands[_numBands - 1].band.setBandType(BandType::UPPER);
416  }
417 
418  reset();

421  template <typename SampleType>
422  void MONSTRCrossover<SampleType>::Process2in2out(SampleType* leftSample,
423  SampleType* rightSample,
424  size_t numSamples) {
425 
426  // If the buffer we've been passed is bigger than our static internal buffer, then we need
427  // to break it into chunks
428  const size_t numBuffersRequired {static_cast<size_t>(
429  std::ceil(static_cast<double>(numSamples) / INTERNAL_BUFFER_SIZE)
430  )};
431 
432  for (size_t bufferNumber {0}; bufferNumber < numBuffersRequired; bufferNumber++) {
433 
434  // Calculate how many samples need to be processed in this chunk
435  const size_t numSamplesRemaining {numSamples - (bufferNumber * INTERNAL_BUFFER_SIZE)};
436  const size_t numSamplesToCopy {
437  std::min(numSamplesRemaining, static_cast<size_t>(INTERNAL_BUFFER_SIZE))
438  };
439 
440  SampleType* const leftBufferInputStart {&leftSample[bufferNumber * INTERNAL_BUFFER_SIZE]};
441  SampleType* const rightBufferInputStart {&rightSample[bufferNumber * INTERNAL_BUFFER_SIZE]};
442 
443  // Populate the band specific buffers, and do the processing
444  for (size_t bandIndex {0}; bandIndex < _numBands; bandIndex++) {
445 
446  // Only do processing no bands are soloed or if this band is soloed
447  if (_numBandsSoloed == 0 || _bands[bandIndex].isSoloed) {
448  std::copy(leftBufferInputStart,
449  &leftBufferInputStart[numSamplesToCopy],
450  _bands[bandIndex].leftBuffer);
451 
452  std::copy(rightBufferInputStart,
453  &rightBufferInputStart[numSamplesToCopy],
454  _bands[bandIndex].rightBuffer);
455 
456  _bands[bandIndex].band.process2in2out(_bands[bandIndex].leftBuffer,
457  _bands[bandIndex].rightBuffer,
458  numSamplesToCopy);
459  }
460  }
461 
462  // Combine the output from each band, and write to output
463  for (size_t sampleIndex {0}; sampleIndex < numSamplesToCopy; sampleIndex++) {
464 
465  // Reset the sample to zero first
466  leftBufferInputStart[sampleIndex] = 0;
467  rightBufferInputStart[sampleIndex] = 0;
468 
469  for (size_t bandIndex {0}; bandIndex < _numBands; bandIndex++) {
470 
471  // Only do processing no bands are soloed or if this band is soloed
472  if (_numBandsSoloed == 0 || _bands[bandIndex].isSoloed) {
473  leftBufferInputStart[sampleIndex] += _bands[bandIndex].leftBuffer[sampleIndex];
474  rightBufferInputStart[sampleIndex] += _bands[bandIndex].rightBuffer[sampleIndex];
475  }
476  }
477  }
478  }
479  }
