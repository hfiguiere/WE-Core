387  template <typename T>
388  void SongbirdFilterModule<T>::Process1in1out(T* inSamples, size_t numSamples) {
389 
390  for (size_t index {0}; index < numSamples; index++) {
391 
392  // Figure out the modulation here. We have two ways to modulation between
393  // two formant filters.
394  // For MODMODE_BLEND we modulation the filter position to blend between the two filters.
395  // For MODMOD_FREQ we set the filter position to 0 so that we're only using filter 1,
396  // and then modulate the freqency of filter 1 between the two vowels
397  double blendFilterPosition {0};
398  const double modAmount {_modulationSource != nullptr ? _modulationSource->getNextOutput(inSamples[index]) : 0};
399  if (_modMode == Parameters::MODMODE_BLEND) {
400  blendFilterPosition = _filterPosition + modAmount;
401  } else {
402  blendFilterPosition = 0;
403  _setVowel1(_calcVowelForFreqMode(modAmount));
404  }
405 
406  // Do the processing for each filter
407  const T originalInput {inSamples[index]};
408  const T filter1Out = _filter1Left.process(originalInput);
409  const T filter2Out = _filter2Left.process(originalInput);
410  const T filterAirOut = _filterAirLeft.process(originalInput);
411 
412  // Write to output, applying filter position and mix level
413  inSamples[index] = (
414  originalInput * (1 - _mix)
415  + filter1Out * (1 - blendFilterPosition) * _mix
416  + filter2Out * blendFilterPosition * _mix
417  + filterAirOut * _airGain * _mix
418  )
419  * _outputGain;
420  }
421  }

423  template <typename T>
424  void SongbirdFilterModule<T>::Process2in2out(T* leftSamples,
425  T* rightSamples,
426  size_t numSamples) {
427 
428  for (size_t index {0}; index < numSamples; index++) {
429 
430  // Figure out the modulation here. We have two ways to modulation between
431  // two formant filters.
432  // For MODMODE_BLEND we modulation the filter position to blend between the two filters.
433  // For MODMOD_FREQ we set the filter position to 0 so that we're only using filter 1,
434  // and then modulate the freqency of filter 1 between the two vowels
435  double blendFilterPosition {0};
436  const double modAmount {
437  _modulationSource != nullptr ? _modulationSource->getNextOutput((leftSamples[index] + rightSamples[index]) / 2) : 0
438  };
439 
440  if (_modMode == Parameters::MODMODE_BLEND) {
441  blendFilterPosition = _filterPosition + modAmount;
442  } else {
443  blendFilterPosition = 0;
444  _setVowel1(_calcVowelForFreqMode(modAmount));
445  }
446 
447  // Do the processing for each filter
448  const T originalLeftIn {leftSamples[index]};
449  const T originalRightIn {rightSamples[index]};
450 
451  const T filter1LeftOut = _filter1Left.process(originalLeftIn);
452  const T filter1RightOut = _filter1Right.process(originalRightIn);
453 
454  const T filter2LeftOut = _filter2Left.process(originalLeftIn);
455  const T filter2RightOut = _filter2Right.process(originalRightIn);
456 
457  const T filterAirLeftOut = _filterAirLeft.process(originalLeftIn);
458  const T filterAirRightOut = _filterAirRight.process(originalRightIn);
459 
460 
461  // Write to output, applying filter position and mix level
462  leftSamples[index] = (
463  originalLeftIn * (1 - _mix)
464  + filter1LeftOut * (1 - blendFilterPosition) * _mix
465  + filter2LeftOut * blendFilterPosition * _mix
466  + filterAirLeftOut * _airGain * _mix
467  )
468  * _outputGain;
469 
470  rightSamples[index] = (
471  originalRightIn * (1 - _mix)
472  + filter1RightOut * (1 - blendFilterPosition) * _mix
473  + filter2RightOut * blendFilterPosition * _mix
474  + filterAirRightOut * _airGain * _mix
475  )
476  * _outputGain;
477  }

486  template <typename T>
487  Vowel SongbirdFilterModule<T>::_calcVowelForFreqMode(double modAmount) {
488  // get the first and second vowels
489  Vowel tempVowel1(getVowelDescription(getVowel1()));
490  Vowel tempVowel2(getVowelDescription(getVowel2()));
491 
492  Vowel retVal(tempVowel1);
493 
494  for (size_t iii {0}; iii < NUM_FORMANTS_PER_VOWEL; iii++) {
495  // Calculate frequency modulation
496  const double freqDelta {tempVowel2[iii].frequency - tempVowel1[iii].frequency};
497  retVal[iii].frequency = tempVowel1[iii].frequency + freqDelta * (_filterPosition + modAmount);
498 
499  // Calculate gain modulation
500  const double gainDelta {tempVowel2[iii].gaindB - tempVowel1[iii].gaindB};
501  retVal[iii].gaindB = tempVowel1[iii].gaindB + gainDelta * (_filterPosition + modAmount);
502  }
503 
504  return retVal;
505  }
