363  template <typename T>
364  void SongbirdFilterModule<T>::Process1in1out(T* inSamples,
365  size_t numSamples) {
366 
367  // If the buffer we've been passed is bigger than our static internal buffer, then we need
368  // to break it into chunks
369  const size_t numBuffersRequired {static_cast<size_t>(
370  std::ceil(static_cast<double>(numSamples) / INTERNAL_BUFFER_SIZE)
371  )};
372 
373  for (size_t bufferNumber {0}; bufferNumber < numBuffersRequired; bufferNumber++) {
374 
375  // Calculate how many samples need to be processed in this chunk
376  const size_t numSamplesRemaining {numSamples - (bufferNumber * INTERNAL_BUFFER_SIZE)};
377  const size_t numSamplesToCopy {std::min(numSamplesRemaining,
378  static_cast<size_t>(INTERNAL_BUFFER_SIZE))};
379 
380  // Get a pointer to the start of this chunk
381  T* const bufferInputStart {&inSamples[bufferNumber * INTERNAL_BUFFER_SIZE]};
382 
383  // Copy the samples we need to process in this chunk into the internal buffers
384  std::copy(bufferInputStart, &bufferInputStart[numSamplesToCopy], _leftOutputBuffer1);
385 
386  std::copy(bufferInputStart, &bufferInputStart[numSamplesToCopy], _leftOutputBuffer2);
387 
388  // Figure out the modulation here. We have two ways to modulation between
389  // two formant filters.
390  // For MODMODE_BLEND we modulation the filter position to blend between the two filters.
391  // For MODMOD_FREQ we set the filter position to 0 so that we're only using filter 1,
392  // and then modulate the freqency of filter 1 between the two vowels
393  double blendFilterPosition {0};
394  const double modAmount {_modulationSource != nullptr ? _modulationSource->getNextOutput(bufferInputStart[0]) : 0};
395  if (_modMode == Parameters::MODMODE_BLEND) {
396  blendFilterPosition = _filterPosition + modAmount;
397  } else {
398  blendFilterPosition = 0;
399  _setVowel1(_calcVowelForFreqMode(modAmount));
400  }
401 
402  // Do the processing for each filter
403  _filters1[Channels::LEFT].process(_leftOutputBuffer1, numSamplesToCopy);
404 
405  _filters2[Channels::LEFT].process(_leftOutputBuffer2, numSamplesToCopy);
406 
407  // Write to output, applying filter position and mix level
408  // always use modFilterPosition as this will take into account any modulation
409  for (size_t iii {0}; iii < numSamplesToCopy; iii++) {
410 
411  // Manually advance the modulation state through the rest of this buffer
412  if (_modulationSource != nullptr && iii != 0) {
413  _modulationSource->getNextOutput(bufferInputStart[iii]);
414  }
415 
416  bufferInputStart[iii] = (
417  bufferInputStart[iii] * (1 - _mix)
418  + _leftOutputBuffer1[iii] * (1 - blendFilterPosition) * _mix
419  + _leftOutputBuffer2[iii] * blendFilterPosition * _mix
420  )
421  * _outputGain;
422  }
423  }

508  template <typename T>
509  Vowel SongbirdFilterModule<T>::_calcVowelForFreqMode(double modAmount) {
510  // get the first and second vowels
511  Vowel tempVowel1(getVowelDescription(getVowel1()));
512  Vowel tempVowel2(getVowelDescription(getVowel2()));
513 
514 
515  Vowel retVal(tempVowel1);
516 
517  for (size_t iii {0}; iii < NUM_FORMANTS_PER_VOWEL; iii++) {
518  // TODO: There's definitely some optimisation to be done below
519 
520  // Calculate frequency modulation
521  const double freqDelta {tempVowel2[iii].frequency - tempVowel1[iii].frequency};
522  retVal[iii].frequency = tempVowel1[iii].frequency + freqDelta * (_filterPosition + modAmount);
523 
524  // Calculate gain modulation
525  const double gainDelta {tempVowel2[iii].gaindB - tempVowel1[iii].gaindB};
526  retVal[iii].gaindB = tempVowel1[iii].gaindB + gainDelta * (_filterPosition + modAmount);
527  }
528 
529  return retVal;
