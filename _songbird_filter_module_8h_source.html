343  template <typename T>
344  void SongbirdFilterModule<T>::Process1in1out(T* inSamples,
345  size_t numSamples) {
346 
347  // If the buffer we've been passed is bigger than our static internal buffer, then we need
348  // to break it into chunks
349  const size_t numBuffersRequired {static_cast<size_t>(
350  std::ceil(static_cast<double>(numSamples) / INTERNAL_BUFFER_SIZE)
351  )};
352 
353  for (size_t bufferNumber {0}; bufferNumber < numBuffersRequired; bufferNumber++) {
354 
355  // Calculate how many samples need to be processed in this chunk
356  const size_t numSamplesRemaining {numSamples - (bufferNumber * INTERNAL_BUFFER_SIZE)};
357  const size_t numSamplesToCopy {std::min(numSamplesRemaining,
358  static_cast<size_t>(INTERNAL_BUFFER_SIZE))};
359 
360  // Get a pointer to the start of this chunk
361  T* const bufferInputStart {&inSamples[bufferNumber * INTERNAL_BUFFER_SIZE]};
362 
363  // Copy the samples we need to process in this chunk into the internal buffers
364  std::copy(bufferInputStart, &bufferInputStart[numSamplesToCopy], _leftOutputBuffer1);
365 
366  std::copy(bufferInputStart, &bufferInputStart[numSamplesToCopy], _leftOutputBuffer2);
367 
368  // Figure out the modulation here. We have two ways to modulation between
369  // two formant filters.
370  // For MODMODE_BLEND we modulation the filter position to blend between the two filters.
371  // For MODMOD_FREQ we set the filter position to 0 so that we're only using filter 1,
372  // and then modulate the freqency of filter 1 between the two vowels
373  double blendFilterPosition;
374  if (_modMode == Parameters::MODMODE_BLEND) {
375  blendFilterPosition = _filterPosition + _modulationSrc;
376  } else {
377  blendFilterPosition = 0;
378  _setVowel1(_calcVowelForFreqMode());
379  }
380 
381  // Do the processing for each filter
382  _filters1[Channels::LEFT].process(_leftOutputBuffer1, numSamplesToCopy);
383 
384  _filters2[Channels::LEFT].process(_leftOutputBuffer2, numSamplesToCopy);
385 
386  // Write to output, applying filter position and mix level
387  // always use modFilterPosition as this will take into account any modulation
388  for (size_t iii {0}; iii < numSamplesToCopy; iii++) {
389  bufferInputStart[iii] = bufferInputStart[iii] * (1 - _mix)
390  + _leftOutputBuffer1[iii] * (1 - blendFilterPosition) * _mix
391  + _leftOutputBuffer2[iii] * blendFilterPosition * _mix;
392  }
393  }
394  }

465  template <typename T>
466  Vowel SongbirdFilterModule<T>::_calcVowelForFreqMode() {
467  // get the first and second vowels
468  Vowel tempVowel1(getVowelDescription(getVowel1()));
469  Vowel tempVowel2(getVowelDescription(getVowel2()));
470 
471 
472  Vowel retVal(tempVowel1);
473 
474  for (size_t iii {0}; iii < NUM_FORMANTS_PER_VOWEL; iii++) {
475  // TODO: There's definitely some optimisation to be done below
476 
477  // Calculate frequency modulation
478  double freqDelta {std::fabs(tempVowel1[iii].frequency - tempVowel2[iii].frequency)};
479  // Invert the delta depending on which value is largest
480  freqDelta *= (tempVowel1[iii].frequency > tempVowel2[iii].frequency) ? -1.0 : 1.0;
481 
482  retVal[iii].frequency = tempVowel1[iii].frequency + freqDelta / 2;
483  retVal[iii].frequency += (freqDelta / 2) * _modulationSrc
484  + (freqDelta / 2) * ((_filterPosition - 0.5) * 2);
485 
486  // Calculate gain modulation
487  double gainDelta {std::fabs(tempVowel1[iii].gaindB - tempVowel2[iii].gaindB)};
488  // Invert the delta depending on which value is largest
489  gainDelta *= (tempVowel1[iii].gaindB > tempVowel2[iii].gaindB) ? -1.0 : 1.0;
490 
491  retVal[iii].gaindB = tempVowel1[iii].gaindB + gainDelta / 2;
492  retVal[iii].gaindB += (gainDelta / 2) * _modulationSrc
493  + (gainDelta / 2) * ((_filterPosition - 0.5) * 2);
494  }
495 
496  return retVal;
497  }
